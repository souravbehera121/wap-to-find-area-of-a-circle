file_name= input("Input filename: ") 
extentions= file_name.split('.')

file_extentions= {
	                  "htm" :"html", 
	                  "py" :"python", 
	                  "cpp" :"C++", 
	                  "c" :"C", 
	                  "java" :"java", 
	                  "css" :"CSS"
	                 } 
output="" 	 
for ext in extentions :
     output = file_extentions.get(ext, " Not found " ) 
print ("The file extension is: " + output)
